Python Integration Example

fetch_samples.py

import requests

# SPARQL Query to fetch ID and Label of all genomic samples
query = """
SELECT ?sample ?sampleLabel WHERE {
  ?sample wdt:P5 wd:Q49. # Instances involving a specific species
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }
}
LIMIT 10
"""

url = "https://rodrigo-test-ciimar.wikibase.cloud/query/sparql"
headers = { 'User-Agent': 'CIIMAR-Agent/1.0', 'Accept': 'application/sparql-results+json' }

response = requests.get(url, params={'query': query}, headers=headers)
data = response.json()

for result in data['results']['bindings']:
    print(f"Sample: {result['sampleLabel']['value']} ({result['sample']['value']})")
